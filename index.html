exports.handler = async function(event, context) { try { const apiKey = process.env.MICROCMS_API_KEY; // APIキーが設定されていない場合のエラーハンドリング if (!apiKey) { return { statusCode: 500, headers: { 'Content-Type': 'application/json' }, body: JSON.stringify({ error: "MICROCMS_API_KEY is not configured in the environment variables." }) }; } // microCMS APIエンドポイント (最新3件を取得) const endpoint = 'https://bonobo.microcms.io/api/v1/news?limit=3'; const response = await fetch(endpoint, { headers: { 'X-MICROCMS-API-KEY': apiKey } }); // APIからのレスポンスが正常でない場合のエラーハンドリング if (!response.ok) { return { statusCode: response.status, headers: { 'Content-Type': 'application/json' }, body: JSON.stringify({ error: `microCMS API error: ${response.statusText}` }) }; } const data = await response.json(); // 取得結果をそのままJSONで返す return { statusCode: 200, headers: { 'Content-Type': 'application/json' }, body: JSON.stringify(data) }; } catch (error) { // 予期せぬエラー（ネットワークエラー等）のハンドリング return { statusCode: 500, headers: { 'Content-Type': 'application/json' }, body: JSON.stringify({ error: "Internal Server Error", message: error.message }) }; } };
